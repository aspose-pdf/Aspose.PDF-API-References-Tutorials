### Παράδειγμα πηγαίου κώδικα για XPS σε PDF χρησιμοποιώντας Aspose.PDF για .NET

```csharp
try
{
	
	// Η διαδρομή προς τον κατάλογο εγγράφων.
	string dataDir = "YOUR DOCUMENT DIRECTORY";

	//Δημιουργήστε το αντικείμενο Instant LoadOption χρησιμοποιώντας την επιλογή φόρτωσης XPS

}

	Console.WriteLine(ex.Message);
}
```
