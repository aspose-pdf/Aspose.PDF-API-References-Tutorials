การสร้างไฮเปอร์ลิงก์ภายในไฟล์ PDF ช่วยให้คุณสร้างลิงก์แบบคลิกได้ซึ่งจะนำผู้ใช้ไปยังหน้าอื่นๆ ในเอกสาร PDF เดียวกัน ด้วย Aspose.PDF สำหรับ .NET คุณสามารถสร้างลิงก์ดังกล่าวได้อย่างง่ายดายโดยทำตามโค้ดต้นฉบับต่อไปนี้:

## ขั้นตอนที่ 1: นำเข้าไลบรารีที่จำเป็น

ก่อนที่คุณจะเริ่มต้น คุณต้องนำเข้าไลบรารีที่จำเป็นสำหรับโครงการ C# ของคุณ นี่คือคำสั่งนำเข้าที่จำเป็น:

```csharp
using Aspose.Pdf;
using Aspose.Pdf.Text;
using Aspose.Pdf.InteractiveFeatures;
```

## ขั้นตอนที่ 2: ตั้งค่าเส้นทางไปยังโฟลเดอร์เอกสาร

 ในขั้นตอนนี้ คุณต้องระบุเส้นทางไปยังโฟลเดอร์ที่คุณต้องการบันทึกไฟล์ PDF ที่ได้ แทนที่`"YOUR DOCUMENT DIRECTORY"` ในโค้ดต่อไปนี้โดยมีเส้นทางจริงไปยังโฟลเดอร์เอกสารของคุณ:

```csharp
string dataDir = "YOUR DOCUMENT DIRECTORY";
```

## ขั้นตอนที่ 3: สร้างอินสแตนซ์ของเอกสาร

 เราจะสร้างอินสแตนซ์ของ`Document` คลาสที่จะแสดงเอกสาร PDF ของเรา นี่คือโค้ดที่เกี่ยวข้อง:

```csharp
Document doc = new Document();
```

## ขั้นตอนที่ 4: เพิ่มหน้าและข้อความด้วยไฮเปอร์ลิงก์

ในขั้นตอนนี้ เราจะเพิ่มหน้าลงในเอกสาร PDF และเพิ่มข้อความที่มีไฮเปอร์ลิงก์ภายใน เราจะกำหนดหน้าเป้าหมายสำหรับแต่ละลิงก์ นี่คือโค้ดที่เกี่ยวข้อง:

```csharp
Page page = doc.Pages.Add();

TextFragment text = new TextFragment("Link to page 7");
LocalHyperlink link = new LocalHyperlink();
link.TargetPageNumber = 7;
text. Hyperlink = link;
page.Paragraphs.Add(text);

text = new TextFragment("Link to page 1");
text. IsInNewPage = true;
link = new LocalHyperlink();
link.TargetPageNumber = 1;
text. Hyperlink = link;
page.Paragraphs.Add(text);
```

## ขั้นตอนที่ 5: บันทึกเอกสารที่อัพเดต

ตอนนี้เรามาบันทึกไฟล์ PDF ที่อัปเดตโดยใช้`Save` วิธีการของ`doc` วัตถุ นี่คือโค้ดที่สอดคล้องกัน:

```csharp
dataDir = dataDir + "CreateLocalHyperlink_out.pdf";
doc.Save(dataDir);
```

### ตัวอย่างโค้ดต้นฉบับสำหรับการสร้างไฮเปอร์ลิงก์ภายในโดยใช้ Aspose.PDF สำหรับ .NET 
```csharp
// เส้นทางไปยังไดเร็กทอรีเอกสาร
string dataDir = "YOUR DOCUMENT DIRECTORY";
// สร้างอินสแตนซ์เอกสาร
Document doc = new Document();
// เพิ่มหน้าเข้าไปยังคอลเลคชันไฟล์ PDF
Page page = doc.Pages.Add();
// สร้างอินสแตนซ์ของ Text Fragment
Aspose.Pdf.Text.TextFragment text = new Aspose.Pdf.Text.TextFragment("link page number test to page 7");
// สร้างอินสแตนซ์ไฮเปอร์ลิงก์ท้องถิ่น
Aspose.Pdf.LocalHyperlink link = new Aspose.Pdf.LocalHyperlink();
// ตั้งค่าหน้าเป้าหมายสำหรับอินสแตนซ์ลิงก์
link.TargetPageNumber = 7;
// ตั้งค่าไฮเปอร์ลิงก์ TextFragment
text.Hyperlink = link;
// เพิ่มข้อความลงในคอลเลกชันย่อหน้าของหน้า
page.Paragraphs.Add(text);
// สร้างอินสแตนซ์ TextFragment ใหม่
text = new TextFragment("link page number test to page 1");
// ควรเพิ่ม TextFragment เหนือหน้าใหม่
text.IsInNewPage = true;
// สร้างอินสแตนซ์ไฮเปอร์ลิงก์ท้องถิ่นอื่น
link = new LocalHyperlink();
// ตั้งค่าหน้าเป้าหมายสำหรับไฮเปอร์ลิงก์ที่สอง
link.TargetPageNumber = 1;
// ตั้งค่าลิงก์สำหรับ TextFragment ที่สอง
text.Hyperlink = link;
// เพิ่มข้อความลงในคอลเลกชันย่อหน้าของวัตถุหน้า
page.Paragraphs.Add(text);    
dataDir = dataDir + "CreateLocalHyperlink_out.pdf";
// บันทึกเอกสารอัพเดต
doc.Save(dataDir);
Console.WriteLine("\nLocal hyperlink created successfully.\nFile saved at " + dataDir);            
```
